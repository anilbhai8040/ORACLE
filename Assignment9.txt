    CREATE TABLE department(
    dept_id NUMBER(3) PRIMARY KEY,
    dept_name VARCHAR2(20)
    );

    CREATE TABLE stud_member(
    roll_no NUMBER(3),
    fname VARCHAR2(15),
    mname VARCHAR2(15),
    sname VARCHAR2(15),
    dept_id NUMBER(3) REFERENCES department(dept_id),
    semester NUMBER(2),
    contact_no NUMBER(10),
    gender CHAR(1) CHECK (gender IN('M','F'))
    );

1. Display the names and contact numbers of all student members.

    SELECT fname,mname,sname,contact_no FROM stud_member;

2. Give the names and roll numbers of all students of Information Technology who are members.

    SELECT s.roll_no,s.fname FROM stud_member s,department d WHERE s.dept_id = d.dept_id and d.dept_name = 'Information Technology';

3. Display names of Departnents whose students are members.

    SELECT distinct(d.dept_name) FROM stud_member s,department d WHERE s.dept_id = d.dept_id;

4. Display names of Departnents for which no students are members.

    SELECT dept_name FROM department WHERE dept_id NOT IN (SELECT DISTINCT dept_id FROM stud_member);

5. Display names of all Departments.

    SELECT dept_name FROM department;

6. Find the number of students of Electrical Department who are members.

    SELECT COUNT(dept_id) AS Total FROM stud_member WHERE dept_id = 2;
    SELECT COUNT(s.dept_id) AS Total FROM stud_member s,department d WHERE s.dept_id = d.dept_id AND d.dept_name = 'Electronical';

7. Display information of student members whose name begins with the letter “A”.

    SELECT *FROM stud_member WHERE fname LIKE 'A%';

8. Display all details of Male members only.

    SELECT *FROM stud_member WHERE gender = 'M';

9. Display data of student members who are curently in semester 3.

    SELECT *FROM stud_member WHERE semester = 3;

10. Display data of student female members in alphabetical order.

    SELECT * FROM stud_member WHERE gender = 'F' ORDER BY fname;
