-- Create the Department table.

    CREATE TABLE Department (
        Dept_id NUMBER(3) PRIMARY KEY,
        Dept_name VARCHAR2(25) NOT NULL,
        Dept_location VARCHAR2(25) NOT NULL
    );

-- Create the Employee table with a Foreign Key reference to Department.

    CREATE TABLE Employee (
        Emp_id NUMBER(5) PRIMARY KEY,
        Emp_Name VARCHAR2(25) NOT NULL,
        DOJ DATE,
        DOB DATE,
        Salary NUMBER(15, 2) DEFAULT 0.0,
        Dept_id NUMBER(3),
        FOREIGN KEY (Dept_id) REFERENCES Department(Dept_id)
    );

1. Display only those records which are work in sales department.

    SELECT e.* FROM Employee e , Department d WHERE e.Dept_id = d.Dept_id AND d.Dept_name = 'Sales';

2. Update salary of employee with 500 Rs/- where dept_id is 20.

    UPDATE Employee SET Salary = Salary + 500 WHERE Dept_id = 20;

3. Display only those employee details whose joining year is 2007.

    SELECT * FROM Employee WHERE EXTRACT(YEAR FROM DOJ) = 2007;

4. Create view for manager which provides details of both tables.

    CREATE OR REPLACE VIEW manager_view AS SELECT e.Emp_id, e.Emp_Name, e.DOJ, e.DOB, e.Salary, d.Dept_name, d.Dept_location FROM Employee e , Department d WHERE e.Dept_id = d.Dept_id;

5. Display sum of salary given to the production department.

    SELECT SUM(e.Salary) AS Total FROM Employee e, Department d WHERE e.Dept_id = d.Dept_id AND d.Dept_name = 'Production';

6. Display only those records where dept_location is Rajkot.

    SELECT e.* FROM Employee e, Department d WHERE e.Dept_id = d.Dept_id AND d.Dept_location = 'Rajkot';

7. Display minimum and maximum salary given to the employee.

    SELECT MIN(Salary) AS Minimum_Salary, MAX(Salary) AS Maximum_Salary FROM Employee;

8. Display name and experience of the employees.

    SELECT Emp_Name, ROUND(MONTHS_BETWEEN(SYSDATE, DOJ) / 12, 2) AS Experience_In_Years FROM Employee;

9. Display the number of days between system date and 1st January 2011.

    SELECT SYSDATE - TO_DATE('01-JAN-2011', 'DD-MON-YYYY') AS Days_Since_Jan2011 FROM DUAL;

10. Display name after converting the first letter of each name to upper case and the rest to lower case.

    SELECT INITCAP(Emp_Name) AS Formatted_Name FROM Employee;

11. Display names and Salary of employee whose name begin with 'A' or 'B'.

    SELECT Emp_Name, Salary FROM Employee WHERE Emp_Name LIKE 'A%' OR Emp_Name LIKE 'B%';

12. Display name of employee and age as on today.

    SELECT Emp_Name,FLOOR(MONTHS_BETWEEN(SYSDATE, DOB) / 12) AS Age FROM Employee;

13. Display the information of employee whose name is greater the 5 character.

    SELECT * FROM Employee WHERE LENGTH(Emp_Name) > 5;

14. Delete the information of whose salary between 5000 to 10000.

    DELETE FROM Employee WHERE Salary BETWEEN 5000 AND 10000;
