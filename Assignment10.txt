1. Count how many orders have made a customer with CustomerName of Smith.

    SELECT customerName,COUNT(customerName) AS total FROM sales GROUP BY customerName HAVING customerName = 'Smith';

2. Find number of unique customers that have ordered from the store.

    SELECT COUNT(DISTINCT(customerName)) AS total_customer FROM sales;

3. Find out total no. of items ordered by all the customers.

    SELECT SUM(orderQuantity) AS Total_Order FROM sales;

4. Find out average number of items per order.

    SELECT AVG(orderQuantity) AS Average_Order FROM sales;
                        OR
    SELECT customerName,AVG(orderQuantity) AS Average_Order FROM sales GROUP BY customerName;

5. Find out the average Order Quantity for all orders with OrderPrice greater than 200

    SELECT AVG(orderQuantity) AS Total_Average_Order FROM sales WHERE orderPrice > 200;

6. Find out what was the minimum price paid for any of the orders.

    SELECT MIN(orderPrice) AS Minimum FROM sales;
                 OR
    SELECT customerName,MIN(orderPrice) AS Minimum FROM sales GROUP BY customerName;

7. Find out the highest OrderPrice from the given sales table

    SELECT MAX(orderPrice) AS MAximum FROM sales;

8. List out unique customers' name only from the table.

    SELECT DISTINCT(customerName) AS CUTOMER_NAME FROM sales;

9. List out name of the customers who have given order in the month of DECEMBER

    SELECT customerName FROM sales WHERE orderDate like '%DEC%';

10. Find out the total amount of money spent for each of the customers.

    SELECT customerName,SUM(orderPrice*orderQuantity) AS TOtal_Price FROM sales GROUP BY customerName;

11. Select all unique customers, who have spent more than 1200 in the store.

    SELECT customerName,SUM(orderPrice*orderQuantity) AS TOtal_Price FROM sales GROUP BY customerName HAVING SUM(orderPrice*orderQuantity)>1200;

12. Select all customers that have ordered more than 5 items in total from all their orders.

    SELECT customerName,SUM(orderQuantity) AS TOtal_Price FROM sales GROUP BY customerName HAVING SUM(orderQuantity)>5 ;

13. Select all customers who have spent more than 1000, after 10/01/2005.

    SELECT customerName, SUM(orderPrice * orderQuantity) AS total_spent FROM sales WHERE orderDate > TO_DATE('2005-01-10','YYYY-MM-DD') GROUP BY customerName HAVING SUM(orderPrice * orderQuantity) > 1000;

14. Select orders in increasing order of order price.

    SELECT customerName,SUM(orderPrice) AS TOtal_Price FROM sales GROUP BY customerName ORDER BY SUM(orderPrice);

15. Select orders in decreasing order of order price.

    SELECT customerName,SUM(orderPrice) AS TOtal_Price FROM sales GROUP BY customerName ORDER BY SUM(orderPrice) DESC;
