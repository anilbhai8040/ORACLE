// Create the following tables.

-- Create the SALESMEN table

    CREATE TABLE SALESMEN (
        SID NUMBER(5) PRIMARY KEY,
        SNAME VARCHAR2(50) NOT NULL,
        SADD VARCHAR2(50),
        SPHONE NUMBER(10)
    );

-- Create the ITEM table

    CREATE TABLE ITEM (
        ITEMNO NUMBER(5) PRIMARY KEY,
        ITEMNAME VARCHAR2(50) NOT NULL,
        PRICE NUMBER(10, 2) CHECK (PRICE > 0)
    );

-- Create the SMAN_ITEM (Sales Transaction) table

-- SID and ITEMNO are foreign keys referencing the parent tables

    CREATE TABLE SMAN_ITEM (
        SID NUMBER(5),
        ITEMNO NUMBER(5),
        QTY NUMBER(5) CHECK (QTY > 0),
        FOREIGN KEY (SID) REFERENCES SALESMEN(SID),
        FOREIGN KEY (ITEMNO) REFERENCES ITEM(ITEMNO)
    );

1. Display details of all those items that have never been sold.

    SELECT i.* FROM ITEM i , SMAN_ITEM si WHERE i.ITEMNO(+) = si.ITEMNO AND si.ITEMNO IS NULL;

    SELECT *FROM ITEM WHERE ITEMNO NOT IN(SELECT i.ITEMNO FROM ITEM i,SMAN_ITEM sm WHERE i.ITEMNO = sm.ITEMNO);

2. Delete all those items that have been sold only once.

    DELETE FROM ITEM WHERE ITEMNO IN (SELECT ITEMNO FROM SMAN_ITEM GROUP BY ITEMNO HAVING COUNT(ITEMNO) = 1);

3. Display all those salesmen who have sale out more than 1000 qty of the same item.

    SELECT *FROM SALESMEN WHERE SID IN (SELECT SID FROM SMAN_ITEM GROUP BY SID HAVING SUM(QTY) > 1000);

4. List all salesmen with their total amount of selling.

    SELECT s.SNAME,NVL(SUM(i.PRICE * si.QTY), 0) AS Total_Sales_Amount FROM SALESMEN s, SMAN_ITEM si, ITEM i WHERE s.SID = si.SID(+) AND si.ITEMNO = i.ITEMNO(+) GROUP BY s.SNAME ORDER BY s.SNAME;

5. Display item wise selling details.

    SELECT i.ITEMNAME,SUM(si.QTY) AS Total_Quantity_Sold,SUM(si.QTY * i.PRICE) AS Total_Revenue FROM ITEM i, SMAN_ITEM si WHERE i.ITEMNO = si.ITEMNO GROUP BY i.ITEMNAME, i.PRICE ORDER BY i.ITEMNAME;
