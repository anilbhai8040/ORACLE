// Create the following table.

    CREATE TABLE client_master(
        clientno NUMBER(3) PRIMARY KEY,
        name VARCHAR2(20),
        address VARCHAR2(20),
        connection_type VARCHAR2(20) NOT NULL,
        area VARCHAR2(20) NOT NULL,
        meter_on_rent char(2) NOT NULL,
        CHECK (name = UPPER(name)),
        CHECK (connection_type IN ('Domestic','Business','Temporary')),
        CHECK (area IN ('Rural','Urban')),
        CHECK (meter_on_rent IN ('Y','N'))
    );
