1. Display the list of customer with department name.  

    SELECT Fname,Lname,dept_name FROM customer;

2. List of customer by salary wise.  

    SELECT *FROM customer ORDER BY salary ASC;
    SELECT *FROM customer ORDER BY salary DESC;

3. Display details of customer whose birth date is in year of ‘1980’.  

    SELECT *FROM customer WHERE birth_date BETWEEN '1-JAN-1980' AND '31-dec-1980';
    SELECT *FROM customer WHERE TO_CHAR(birth_date, 'YYYY') = '1980';

4. Display details of customer whose mobile no start with ’99..’series.  

    SELECT *FROM customer WHERE mob_no LIKE '99%';

5. Display name and depart name of customer whose city starting from ‘M’.  

    SELECT Fname, Lname ,dept_name FROM customer WHERE city LIKE 'm%';

6. Display average salary of customer.  

    SELECT AVG(salary) AS average_salary FROM customer;

7. Display minimum salary in each department.  

    SELECT dept_name, MIN(salary) AS min_salary FROM customer GROUP BY dept_name;

8. Display count of customer having salary greater than 3000.  

    SELECT COUNT(*) AS count_high_salary FROM customer WHERE salary > 3000;

9. List name of customer having ‘A’ as 2nd letter in first name.  

    SELECT * FROM customer WHERE Fname LIKE '_A%';
