    // First create the PRIMARY KEY table

        CREATE TABLE dept(
        deptNo NUMBER(3) PRIMARY KEY,
        dname VARCHAR2(15),
        location VARCHAR2(15)
        );

    // Second create foreign key table

        CREATE TABLE emp(
        emp_id NUMBER(3),
        name VARCHAR2(15),
        city VARCHAR2(10),
        job VARCHAR2(10),
        salary NUMBER(6),
        joinDate DATE,
        commision NUMBER(7,2),
        deptNo NUMBER(3) REFERENCES dept(deptNo)
        );

1. Display detailed of employee joined in the month of ‘MARCH’.

    SELECT *FROM emp WHERE joinDate LIKE '%MAR%';
    SELECT * FROM emp WHERE TO_CHAR(join_date, 'MON') = 'MAR';

2. Display details of employee whose city is not ‘DELHI’ or ‘PUNE’.

    SELECT *FROM emp WHERE city NOT IN ('delhi','pune');
    
3. Display details of employee joined after ‘1st January 1990’. 

    SELECT *FROM emp WHERE joinDate > '1-jan-1990';
    SELECT *FROM emp WHERE join_date > TO_DATE('01-JAN-1990', 'DD-MON-YYYY');

4. Display details of employee whose is ‘MANAGER’ or ‘OFFICER’. 

    SELECT e.* FROM emp e,dept d WHERE d.deptNo = e.deptNo AND d.dname IN ('manager','officer');

5. Display details of employee in ascending order of name. 

    SELECT *FROM emp ORDER BY name ASC; // default is ASC

6. Display second highest salary of employee. 

    SELECT MAX(salary) AS second_highest_salary FROM emp WHERE salary < (SELECT MAX(salary) FROM emp);

7. Display details of employee whose joining date is in month of ‘FEB’. 

    SELECT *FROM emp WHERE joinDate NOT LIKE '%FEB%';
    SELECT *FROM emp WHERE TO_CHAR(join_date, 'MON') = 'FEB';

8. List details of employee not working in ‘ACCOUNT’ department. 

    SELECT e.* FROM emp e,dept d WHERE d.deptNo = e.deptNo AND d.dname NOT 'accountant';

9. Display the list of employee whose post is ‘PROGRAMMER’ or salary >= 15000. 

    SELECT *FROM emp WHERE job = 'programmer' OR salary >= 15000;

10. Count the number of employee in each department.

    SELECT d.dname , COUNT(e.emp_id) AS total_emp FROM emp e,dept d WHERE d.deptNo = e.deptNo;
    SELECT d.dname , COUNT(e.emp_id) AS total_emp FROM emp e,dept d WHERE d.deptNo = e.deptNo GROUP BY d.dname;
